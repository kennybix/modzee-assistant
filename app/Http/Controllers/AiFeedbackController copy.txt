<?php

namespace App\Http\Controllers;

use App\Http\Requests\AiFeedbackRequest;
use App\Models\AiFeedback;
use App\Models\AiLog;
use Illuminate\Http\JsonResponse;
use Illuminate\Support\Facades\Auth;

class AiFeedbackController extends Controller
{
    public function store(AiFeedbackRequest $request): JsonResponse
    {
        try {
            $logId = $request->input('response_id');
            $rating = $request->input('rating');
            $comment = $request->input('comment');
            
            // Check if the log exists
            $log = AiLog::findOrFail($logId);
            
            // Create feedback
            AiFeedback::create([
                'ai_log_id' => $logId,
                'user_id' => Auth::id(),
                'rating' => $rating,
                'comment' => $comment,
            ]);
            
            return response()->json([
                'message' => 'Feedback recorded successfully',
                'status' => 'success'
            ]);
        } catch (\Exception $e) {
            return response()->json([
                'error' => 'Failed to record feedback',
                'message' => $e->getMessage()
            ], 500);
        }
    }

    public function getFeedbackStats(): JsonResponse
    {
        $stats = AiLog::getFeedbackStats();
        
        return response()->json($stats);
    }
}
